//Number 8: This program will generate a multiplication table

#include <iostream>
#include <conio.h>
using namespace std;
int main()
{
int n1;

    cout<<"Enter a Number:" <<endl;
    cin>>n1;
    cout<<"Multiplication Table of:  " <<n1 <<endl;
    cout<<n1<<"*1=" <<n1*1 <<endl; 
    cout<<n1<<"*2=" <<n1*2 <<endl; 
    cout<<n1<<"*3=" <<n1*3 <<endl;
    cout<<n1<<"*4=" <<n1*4 <<endl;
    cout<<n1<<"*5=" <<n1*5 <<endl;
    cout<<n1<<"*6=" <<n1*6 <<endl;
    cout<<n1<<"*7=" <<n1*7 <<endl;
    cout<<n1<<"*8=" <<n1*8 <<endl;
    cout<<n1<<"*9=" <<n1*9 <<endl;
    cout<<n1<<"*10=" <<n1*10 <<endl;
    
    _getch();
    return 0;
}
